cmake_minimum_required(VERSION 3.29.2)

set(CMAKE_CXX_STANDARD 20)

project(OrderBook)

# wxwidgets build

    src/simple_orderbook.cpp
    src/order.cpp
    src/application.cpp
    src/main_frame.cpp
)

target_link_libraries(${PROJECT_NAME} wx::net wx::core wx::base)
